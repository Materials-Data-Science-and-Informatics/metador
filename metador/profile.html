metador.profile

Metador profile management.

This module takes care of loading profiles and referenced JSON Schemas, assembling them, returning them and providing functions to match files to schemas.

Schemas must be valid according to JSON Schema Draft 7, Profiles must be valid according to the Profile Schema.

Profile.load_profiles must be called with the profile directory as argument at the start of the application.

 72def schema_bfs(
 73    name: str,
 74    func: Callable[[str, UnsafeJSON], Any],
 75    get_succ: Callable[[str], UnsafeJSON],
 76) -> None:
 77    """
 78    Perform BFS on the graph schemas and their successors (referenced schemas).
 79
 80    Takes a schema name, a function to call for each schema, and a successor function
 81    that returns a schema given its name (without fragment part).
 82    """
 83    vis = set(name)
 84    q: Deque[str] = collections.deque()
 85    q.append(name)
 86    while not len(q) == 0:
 87        curr_name = q.popleft()
 88        curr_schema = get_succ(curr_name)
 89        if curr_name in vis:
 90            continue
 91        vis.add(curr_name)
 92        func(curr_name, curr_schema)
 93        for name in referenced_schemas(curr_schema):
 94            q.append(name)

 97def load_schema_refs(
 98    name: str, schema: UnsafeJSON, extra_schemas: Dict[str, UnsafeJSON]
 99) -> Set[str]:
100    """
101    Given a JSON schema dict, recursively load referenced schemas.
102
103    This means, make sure that all referenced schemas are in the cache.
104    Returns all direct and transitive $refs.
105    """
106    ret = set()
107
108    def collect_ref(name: str, _: UnsafeJSON) -> None:
109        ret.add(name)
110
111    def get_next_schema(name: str) -> UnsafeJSON:
112        return Profile.schema_content(name, extra_schemas)
113
114    schema_bfs(name, collect_ref, get_next_schema)
115    return ret

133    """
134    Class holding a profile representation from a loaded *.profile.json file.
135
136    It also includes references to all relevant JSON Schemas.
137    """

139    title: str
140    """Human-readable title of dataset profile."""
141
142    description: Optional[str] = ""
143    """Human-readable description of dataset profile."""
144
145    schemas: Dict[str, UnsafeJSON] = {}
146    """Assembled collection of included schemas."""
147
148    patterns: List[PatternSchema] = []
149    """Mappings regex -> schema (name of schema is resolved to corresp. schema)."""
150
151    rootSchema: str
152    """Schema to use for the dataset itself (not bound to a file)."""
153
154    fallbackSchema: str
155    """Schema to use for files not matching any pattern."""
156
157    def get_schema_for(self, filename: Optional[str]) -> UnsafeJSON:
158        """
159        Return first schema where pattern completely matches filename.
160
161        Otherwise, returns the fallback schema. If no path given, returns the root schema.
162        """
163        if filename is None:
164            return self.schemas[self.rootSchema]
165        for pat in self.patterns:
166            pmatch = re.match(pat.pattern, filename, re.IGNORECASE)
167            if pmatch and pmatch.group(0) == filename:
168                return self.schemas[pat.useSchema]
169        return self.schemas[self.fallbackSchema]
170
171    @classmethod
172    def get_schema_json(
173        cls, filename: str, force_reload: Optional[bool] = False
174    ) -> UnsafeJSON:
175        """
176        If a schema with that name is not loaded yet, load, validate and cache.
177
178        Return schema on success. Set `force_reload=True` to force a reload from file.
179        """
180        global _schemas_json
181
182        if not force_reload and filename in _schemas_json:
183            return _schemas_json[filename]
184
185        # not cached -> load from file/URL
186        schema = None
187        try:
188            schema = load_json(expand_schema_path(filename))
189        except JSONDecodeError:
190            critical_exit("Cannot parse schema file '{filename}'!")
191
192        if validate_json(schema, _JSONSCHEMA_SCHEMA) is not None:
193            critical_exit(f"{filename} is not a valid Draft 7 JSON Schema!")
194
195        _schemas_json[filename] = schema  # cache for next access
196        return schema

198    def save(self, filepath: Path) -> None:
199        """Serialize the current state into a file."""
200        save_json(self, filepath)

202    @classmethod
203    def get_profile_json(
204        cls, filename: str, force_reload: Optional[bool] = False
205    ) -> Mapping[str, UnsafeJSON]:
206        """
207        Given the filename of a profile, load, validate cache and return it if not loaded yet.
208
209        If loaded (and not forcing reload), returns cached version.
210        This is only for loading the "raw" profile. Use 'load' to get the assembled one.

212        global _profiles_json
213

215            return _profiles_json[filename]

225        return content

234    @classmethod
235    def schema_content(
236        cls, filename: str, embedded: Optional[Dict[str, UnsafeJSON]] = None
237    ) -> UnsafeJSON:
238        """
239        Return the schema referenced by provided filename.
240
241        Also handles the special filenames for the trivial true/false schema,
242        for consistent treatment.
243        """
244        if filename == TRIV_TRUE:
245            return True
246        elif filename == TRIV_FALSE:
247            return False
248        elif embedded and filename in embedded:
249            return embedded[filename]
250        else:
251            return cls.get_schema_json(filename)

253    @classmethod
254    def load(cls, filename: str):
255        """
256        Construct a self-contained profile from file (filename relative to profile dir).
257
258        Embeds referenced schemas and produces a kind-of "normal form".
259        """
260        profile: Mapping[str, Any] = cls.get_profile_json(filename)
261
262        title: str = profile["title"] if profile["title"] is not None else filename
263
264        description: str = ""
265        if "description" in profile:
266            description = profile["description"]
267
268        rootSchema: str = cls.schema_filename(profile["rootSchema"])
269        fallbackSchema: str = cls.schema_filename(profile["fallbackSchema"])
270
271        schemas: Dict[str, UnsafeJSON] = {TRIV_TRUE: True, TRIV_FALSE: False}
272
273        toEmbed: Set[str] = set()  # additional referenced schemas that must be embedded
274
275        # first copy the directly embedded ones
276        if "schemas" in profile:  # optional embedded schemas
277            for embName, embSchema in profile["schemas"].items():
278                schemas[embName] = embSchema
279
280        # collect refs from inside the embedded schemas
281        for embName, embSchema in schemas.items():
282            toEmbed = toEmbed.union(load_schema_refs(embName, embSchema, schemas))
283
284        # need to also add root and fallback schemas
285        schemas[rootSchema] = cls.schema_content(rootSchema, schemas)
286        schemas[fallbackSchema] = cls.schema_content(fallbackSchema, schemas)
287        # collect their refs, too
288        toEmbed = toEmbed.union(
289            load_schema_refs(rootSchema, schemas[rootSchema], schemas)
290        )
291        toEmbed = toEmbed.union(
292            load_schema_refs(fallbackSchema, schemas[fallbackSchema], schemas)
293        )
294
295        patterns = []
296        for pat in profile["patterns"]:
297            # add referenced schema from pattern if not yet added
298            schemaref: str = cls.schema_filename(pat["useSchema"])
299            schemaval: UnsafeJSON = cls.schema_content(schemaref, schemas)
300            schemas[schemaref] = schemaval
301            # add regex entry
302            patterns.append(PatternSchema(pattern=pat["pattern"], useSchema=schemaref))
303            # collect $refs
304            toEmbed = toEmbed.union(
305                load_schema_refs(schemaref, schemas[schemaref], schemas)
306            )
307
308        # do not overwrite already embedded ones
309        toEmbed = toEmbed.difference(schemas.keys())
310        # embed missing referenced schemas
311        for schema_name in toEmbed:
312            schemas[schema_name] = cls.schema_content(schema_name, schemas)
313
314        return cls(
315            title=title,
316            description=description,
317            rootSchema=rootSchema,
318            fallbackSchema=fallbackSchema,
319            schemas=schemas,
320            patterns=patterns,
321        )

323    @classmethod
324    def load_profiles(cls, profile_dir: Path) -> None:
325        """
326        Load and validate raw JSON profiles from the configured directory.
327
328        Also loads and validates the referenced JSON Schemas.
329        """
330        global _PROFILE_DIR
331        _PROFILE_DIR = profile_dir
332
333        if not _PROFILE_DIR.is_dir():
334            critical_exit(f"Profile directory {_PROFILE_DIR} does not exist!")
335
336        log.info(f"Loading profiles (*{PROFILE_SUF}) from '{_PROFILE_DIR}'")
337
338        pr_filenames = list(map(lambda x: x.name, _PROFILE_DIR.iterdir()))
339        pr_filenames = list(
340            filter(lambda x: str(x).endswith(PROFILE_SUF), pr_filenames)
341        )
342        if len(pr_filenames) == 0:
343            critical_exit(f"No profiles (*{PROFILE_SUF}) found in {_PROFILE_DIR}!")
344
345        for pr_filename in pr_filenames:
346            name: str = re.sub(PROFILE_SUF + "$", "", str(pr_filename))
347            profile = cls.load(pr_filename)
348            _profiles[name] = profile
349
350    @classmethod
351    def get_profiles(cls) -> List[str]:
352        """
353        Return a list a available profiles.
354
355        (works only after `Profile.load_profiles` was called to initialize)
356        """
357        return list(sorted(_profiles.keys()))

359    @classmethod
360    def get_profile(cls, name: str) -> Profile:
361        """
362        Return a profile given its name (or raises error if no such profile).
363
364        (Works only after `Profile.load_profiles` was called to initialize)
365        """
366        return _profiles[name]
