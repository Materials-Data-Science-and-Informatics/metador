218    def compute_checksum(self, filename: str) -> bool:
219        """Run checksum tool on file and assign checksum."""
220        filepath = self._upload_filepath(filename)
221
222        if filename not in self.files:
223            log.error(
224                f"Cannot compute checksum for {filename}, no such file in dataset!"
225            )
226            return False
227
228        try:
229            self.files[filename].checksum = util.hashsum(filepath, self.checksumAlg)
230        except FileNotFoundError:
231            log.error(f"File {filepath} not found, cannot compute hashsum!")
232            return False
233        except ValueError:
234            log.error(f"Hashing algorithm {self.checksumAlg} is not supported!")
235            return False
236
237        self.save()
238        return True

240    def rename_file(self, name: str, new_name: str) -> bool:
241        """
242        Try to rename an existing file to a new free filename.
243
244        Returns True on success, i.e. file and its data exists and new name is free.
245        """
246        if name not in self.files:
247            log.warning(f"Cannot rename non-existing file {name} from {self.id}")
248            return False
249
250        if name == new_name:
251            log.info(f"Rename {name}->{new_name} in {self.id}: nothing to do.")
252            return True
253
254        if new_name in self.files:
255            log.warning(f"Can't rename {name}->{new_name} in {self.id}, target exists!")
256            return False
257
258        oldpath = self._upload_filepath(name)
259        newpath = self._upload_filepath(new_name)
260
261        if not oldpath.is_file():
262            msg = f"File {oldpath} does not exists, although referenced!"
263            log.error(msg)
264            raise FileNotFoundError(msg)
265        if newpath.is_file():
266            msg = f"File {newpath} exists, but not referenced! Abort rename."
267            log.error(msg)
268            raise FileExistsError(msg)
269
270        oldpath.rename(newpath)
271        self.files[new_name] = self.files[name]
272        del self.files[name]
273
274        self.save()
275        return True

279    def complete(self) -> Tuple[Optional[Path], Dict[str, str]]:
280        """
281        Validate metadata and check that all checksums are present.
282
283        On success, remove files and data from 'staging' directory
284        and produce the directory in the 'complete' directory. Return path.
285        """
286        global _datasets
287
288        val_errors = self.validate_dataset()
289        if len(val_errors) != 0:
290            log.error(f"Cannot complete dataset, validation failed: {val_errors}")
291            return (None, val_errors)
292
293        missing_checksums = {}
294        for file, dat in self.files.items():
295            if dat.checksum is None:
296                missing_checksums[file] = f"File checksum is missing: {file}"
297        if len(missing_checksums) != 0:
298            return (None, missing_checksums)
299
300        upload_dir: Final[Path] = self._upload_dir()
301        target_dir: Final[Path] = self._target_dir()
302
303        # move the files
304        upload_dir.rename(target_dir)
305
306        # produce the metadata files
307        if self.rootMeta is not None:
308            with open(target_dir / METADATA_SUF, "w") as outfile:
309                json.dump(self.rootMeta, outfile, indent=2)
310                outfile.flush()
311
312        for filename, dat in self.files.items():
313            if dat.metadata is not None:
314                with open(target_dir / (filename + METADATA_SUF), "w") as outfile:
315                    json.dump(dat.metadata, outfile, indent=2)
316                    outfile.flush()
317
318        # create a checksum file (e.g. sha256sums.txt)
319        with open(target_dir / (self.checksumAlg + "sums.txt"), "w") as outfile:
320            for name, dat in sorted(self.files.items()):
321                outfile.write(f"{name}  {dat.checksum}\n")
322                outfile.flush()
323
324        # create json for remaining info that might be useful
325        with open(target_dir / "dataset.json", "w") as outfile:
326            dsinfo = DatasetInfo(
327                creator=self.creator, created=self.created, profile=self.profile
328            )
329            json.dump(json.loads(dsinfo.json()), outfile, indent=2)
330            outfile.flush()
331
332        # TODO: this is the point where we could store the profile as  e.g. dirschema
333
334        del _datasets[self.id]
335
336        # delete persistence file (we produced all derivative files already)
337        self._persist_filename(self.id).unlink()
338
339        log.info(f"Dataset {self.id} completed.")
340        return (target_dir, {})

342    def delete(self) -> None:
343        """Delete information about this dataset and all its files IRREVERSIBLY."""
344        upload_dir: Final[Path] = self._upload_dir()
345        for uploaded_file in upload_dir.glob("*"):
346            uploaded_file.unlink()
347        upload_dir.rmdir()
348
349        self._persist_filename(self.id).unlink()
350
351        del _datasets[self.id]
352
353        log.info(f"Data of {self.id} deleted.")

355    @classmethod
356    def load(cls, ds_id: UUID) -> Optional[Dataset]:
357        """Load a dataset from the corresponding serialization file."""
358        persist_file = Dataset._persist_filename(ds_id)
359        if not persist_file.is_file():
360            log.error(f"Failed loading, {str(persist_file)} not found.")
361            return None
362
363        ds = Dataset.parse_file(Dataset._persist_filename(ds_id))
364        if len(ds.files) == 0 and not ds._upload_dir().is_dir():
365            log.warning(f"Upload dir {ds_id} not found! This is odd. Will create one.")
366            ds._upload_dir().mkdir()
367
368        # check that for each entry the files do exist
369        for fname in ds.files.keys():
370            if not ds._upload_filepath(fname).is_file():
371                # this really should not happen that files just vanish... real exception!
372                msg = f"File {fname} referenced in dataset, but not existing!"
373                log.error(msg)
374                raise FileNotFoundError(msg)
375
376        # for each additional file without an entry, create one
377        for filepath in ds._upload_dir().glob("*"):
378            if filepath.name not in ds.files:
379                ds.files[filepath.name] = FileInfo()
380
381        return ds

383    @classmethod
384    def create(cls, profile: Profile, creator: Optional[OrcidStr] = None) -> Dataset:
385        """Create a new dataset and its directory + persistence file, return it."""
386        now = datetime.now()
387        tdiff = timedelta(hours=conf().metador.incomplete_expire_after)
388        ds = cls(
389            id=uuid1(),
390            creator=creator,
391            created=now,
392            expires=now + tdiff,
393            profile=profile,
394            checksumAlg=conf().metador.checksum,
395        )
396
397        ds._upload_dir().mkdir()  # create underlying upload directory
398        ds.save()  # create underlying file
399        _datasets[ds.id] = ds  # register in loaded list
400        log.info(f"New dataset {ds.id} created.")
401        return ds

403    @classmethod
404    def load_datasets(cls) -> None:
405        """Load datasets for which a persistence file exists."""
406        global _datasets
407        cls._prepare_dirs()
408        for ds_id in cls._find_staging_datasets():
409            log.debug(f"Loading dataset {ds_id} from file.")
410            ds = Dataset.load(ds_id)
411            if ds is not None:
412                _datasets[ds_id] = ds

415    def cleanup_datasets(cls) -> None:
416        """Remove expired datasets from cache and delete the files."""
417        t = datetime.today()
418        log.info("Cleaning up expired datasets...")
419        expired = [v for v in _datasets.values() if v.is_expired(t)]
420        for ds in expired:
421            ds.delete()

448        """Create directory structure for datasets at location specified in config."""

450        if not data_dir.is_dir():

458    def _find_staging_datasets(cls) -> List[UUID]:
459        """Get UUIDs of datasets with serialized data in the staging directory."""
460        files = list(cls._staging_dir().glob("*" + DATASET_SUF))
