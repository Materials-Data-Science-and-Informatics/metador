metador.orcid.auth

Module for ORCID authentication, implemented based on the official integration guide.

The guide can be found here: https://info.orcid.org/documentation/api-tutorials/api-tutorial-get-and-authenticated-orcid-id/

This module performs the 3-legged auth and then exchanges the authenticated ORCID for a plain old user session. Common wisdom says that JWT should not be used for sessions. Furthermore, the crypto makes everything more complicated. Sessions via cookies are fine.

This module supports the official production+sandbox servers and the mock server.

This module does not directly rely on Metador-specifics to simplify reuse.

112def load_allowlist(filename: Path) -> List[OrcidStr]:
113    """
114    Load list of ORCIDs that should be allowed to sign in.
115
116    Assumes that the provided path to the allowlist file exists
117    (should be checked before).
118    """
119    lines: List[str] = []
120    with open(filename, "r") as file:
121        lines = file.readlines()
122
123    stripped = list(map(str.strip, lines))
124    nonempty = list(filter(lambda x: x != "", stripped))
125    noncommented = list(filter(lambda x: x[0] != "#", nonempty))
126    invalid = list(filter(lambda oid: not re.match(ORCID_PAT, oid), noncommented))
127    if len(invalid) > 0:
128        critical_exit(f"Invalid ORCIDs in allow list: {invalid}")
129    return noncommented

173    def orcid_server_pref(self) -> str:
174        """Call the actual prefix function with params (convenience wrapper)."""
175        sandbox = self.orcid_conf.sandbox
176        use_fake = self.orcid_conf.use_fake
177        return orcid_server_pref(sandbox, self.site_prefix if use_fake else None)

179    def userauth_url(self, state: Optional[str] = None) -> str:
180        """
181        Construct an URL for login via ORCID for the user to click on.
182
183        For this to work, you must first register your Metadir instance in
184        the developer tools in the ORCID account.
185        Input: -
186        Output: URL for frontend use.
187        """
188        auth_url = self.orcid_server_pref() + "/authorize"
189        auth_url += "?response_type=code&scope=/authenticate"
190
191        orcid_cid = self.orcid_conf.client_id
192        redir = orcid_redir(self.site_prefix)
193        opt_state = f"&state={state}" if state else ""
194
195        return f"{auth_url}{opt_state}&client_id={orcid_cid}&redirect_uri={redir}"

197    async def redeem_code(self, code: str) -> OrcidBearerToken:
198        """
199        Swap code received from ORCID server against a authenticated ORCID token.
200
201        Input: the 6-digit code from ORCID.
202        Output: Bearer token from ORCID certifying that the user logged into ORCID.
203        """
204        hdrs = {
205            "Accept": "application/json",
206            "Content-Type": "application/x-www-form-urlencoded",
207        }
208        dat = {
209            "client_id": self.orcid_conf.client_id,
210            "client_secret": self.orcid_conf.client_secret,
211            "grant_type": "authorization_code",
212            "code": code,
213            "redirect_uri": orcid_redir(self.site_prefix),
214        }
215
216        httpx_client_args = {}
217        if self.orcid_conf.use_fake:
218            httpx_client_args["verify"] = False
219        async with httpx.AsyncClient(**httpx_client_args) as client:
220            r = await client.post(
221                self.orcid_server_pref() + "/token",
222                headers=hdrs,
223                data=dat,
224            )
225        return OrcidBearerToken.parse_obj(r.json())
226
227    async def revoke_token(self, tok: OrcidBearerToken) -> bool:
228        """Invalidate an access token. Return success value."""
229        hdrs = {"Accept": "application/json"}
230        dat = {
231            "client_id": self.orcid_conf.client_id,
232            "client_secret": self.orcid_conf.client_secret,
233            "token": tok.access_token,
234        }
235        httpx_client_args = {}
236        if self.orcid_conf.use_fake:
237            httpx_client_args["verify"] = False
238        async with httpx.AsyncClient(**httpx_client_args) as client:
239            r = await client.post(
240                self.orcid_server_pref() + "/revoke",
241                headers=hdrs,
242                data=dat,
243            )
244        return 200 <= r.status_code < 300

248    def new_session(self, tok: OrcidBearerToken) -> SessionID:
249        """
250        Generate a new session with timestamp from given ORCID token.
251
252        Store it, return the SessionID.
253        """
254        session_id = SessionID(secrets.token_urlsafe(32))  # unique with huge likelihood

256        now = datetime.today()
257        self.sessions[session_id] = Session(
258            created=now,
259            expires=now + timedelta(minutes=self.orcid_conf.auth_expire_after),

261            user_name=tok.name,

267    def persist_file_path(self) -> Path:
268        """Return file path for session serialization (if persist_dir is set)."""
269        assert self.persist_dir is not None
270        return self.persist_dir / self.PERSIST_FILE

272    def load_sessions(self) -> None:
273        """
274        Try to load sessions (assuming persist_dir is set).
275
276        This is called by `init_auth`, which is to be called on startup.
277        """
278        fname = self.persist_file_path()
279
280        if not fname.is_file():
281            log.debug("No serialized session file.")
282            return
283
284        log.info(f"Restoring serialized sessions from {fname}")
285        self.sessions = Sessions.parse_file(fname).__root__
286        self.cleanup_sessions()
287
288    def dump_sessions(self) -> None:
289        """
290        Dump current sessions to file (assuming persist_dir is set).
291
292        This should be called on shutdown of the app.
293        """
294        fname = self.persist_file_path()
295        log.info(f"Serializing sessions to {fname}")
296        serialized = Sessions(__root__=self.sessions).json()
297        with open(fname, "w") as file:
298            file.write(serialized)
299            file.flush()
300
301    def lookup_session(self, session_id: Optional[SessionID]) -> Optional[Session]:
302        """
303        Return session, if it is still valid.
304
305        Otherwise, the session is removed from list (expired, not allowed, etc.).
306        """
307        if session_id is None or session_id == "":  # no session id passed
308            return None
309
310        sess_id = SessionID(session_id)
311
312        if sess_id not in self.sessions:  # session id not in list
313            log.debug(f"Ignoring invalid session {sess_id}.")
314            return None
315
316        session = self.sessions[sess_id]
317
318        if session.is_expired():
319            log.debug(f"Removing expired session {sess_id}.")
320            del self.sessions[sess_id]
321            return None
322
323        # also check the allow list, it might have changed (e.g. server restart)
324        if self.allow_list is not None and session.orcid not in self.allow_list:
325            log.debug(f"Removing session {sess_id} by not allowed user.")
326            del self.sessions[sess_id]
327            return None
328
329        return session
330
331    def cleanup_sessions(self) -> None:
332        """Remove expired sessions."""
333        t = datetime.today()
334        self.sessions = {k: v for k, v in self.sessions.items() if not v.is_expired(t)}
